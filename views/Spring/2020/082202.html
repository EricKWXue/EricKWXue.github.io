# Spring boot 日志最佳实践

  • 选用日志门面模式框架slf4j，内部采用logback框架
  • 分环境进行配置，不同环境对日志的处理方式不同

application.yml

server:
  port: 9090
spring:
  application:
    name: ericapp
  mandatory-file-encoding: UTF-8
  http:
    encoding:
      enabled: true
      charset: utf-8
  profiles:
    active: dev
## 选用开发环境dev，此时application-dev.yml的内容会覆盖application.yml
logging:
  config: classpath:logback-spring.xml

logback-spring.xml

<?xml version="1.0" encoding="UTF-8"?>
<configuration>
    <include resource="org/springframework/boot/logging/logback/base.xml"/>
    <!-- prod文件路径 -->
    <property name="LOG_PATH" value="/home/server/web/logs" />

    <!-- 开发环境 -->
    <springProfile name="dev">
        <!--自己的包打印debug信息，第三方的包打印info信息-->
        <logger name="com.eric" level="debug"/>
        <!--只输出到控制台, 采用base.xml中配置好的CONSOLE-->
        <root level="INFO">
            <appender-ref ref="CONSOLE" />
        </root>
    </springProfile>

    <!-- 生产环境 -->
    <springProfile name="prod">
        <appender name="FILE" class="ch.qos.logback.core.rolling.RollingFileAppender">
            <!-- 正在记录的日志文档的路径及文档名 -->
            <file>${LOG_PATH}/warn.log</file>
            <encoder>
                <!--日志文档输出格式, 采用base.xml中配置好的格式-->
                <pattern>${FILE_LOG_PATTERN}</pattern>
            </encoder>
            <!-- 日志记录器的滚动策略，按日期，按大小记录 -->
            <rollingPolicy class="ch.qos.logback.core.rolling.SizeAndTimeBasedRollingPolicy">
                <cleanHistoryOnStart>false</cleanHistoryOnStart>
                <fileNamePattern>${LOG_PATH}/warn.%d{yyyy-MM-dd}.%i.log}</fileNamePattern>
                <!--每个日志文件的最大大小-->
                <maxFileSize>15MB</maxFileSize>
                <!--日志文档保留天数-->
                <maxHistory>7</maxHistory>
            </rollingPolicy>
        </appender>

        <!--自己的包打印info信息，第三方的包打印warn信息-->
        <logger name="com.eric" level="info"/>
        <root level="WARN">
            <appender-ref ref="FILE" />
        </root>
    </springProfile>

</configuration>
